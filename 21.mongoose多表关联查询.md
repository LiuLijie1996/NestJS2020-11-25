## 设置表规则

### article

```js
const { Schema } = require("mongoose");
let mongoose = require("./db");

// 文章
let ArticleSchema = mongoose.Schema({
    title: {
        type: String,
        unique: true,
    },
    // 新闻分类id
    cid: {
        type: Schema.Types.ObjectId,// id类型
    },
    // 用户id
    author_id: {
        type: Schema.Types.ObjectId,// id类型
    },
    // 用户名
    author_name: {
        type: String,
    },
    // 新闻简介
    descripton: String,
    // 新闻内容
    content: String,
});
module.exports = mongoose.model("Article", ArticleSchema, "article");
```

### articlecate

```js
let mongoose = require("./db");

// 文章分类
let ArticleCateSchema = mongoose.Schema({
   title:{
       type: String,
       unique: true,
   },
   descripton: String,
   content: String,
   add_time:{
       type: Date,
   },
});
module.exports = mongoose.model("ArticleCate", ArticleCateSchema, "articlecate");
```

### user

```js
let mongoose = require("./db");

// 用户
let UserSchema = mongoose.Schema({
    username: {
        type: String,
        unique: true,
    },
    password: String,
    name: String,
    age: Number,
    sex:String,
    tel: Number,
    status: {
        type: Number,
        default: 1,
    },
});
module.exports = mongoose.model("User", UserSchema, "user");
```



### 多表关联查询

```js
let Article = require("./model/article");

// 多表关联查询
Article.aggregate(
    [
        {
            $lookup: {
                from: 'articlecate',// 需要关联的表的名称
                localField: "cid",// article表的 cid 字段
                foreignField: "_id",// articlecate表的 _id 字段
                as: "cate",
            }
        },
        {
            $lookup: {
                from: 'user',// 需要关联的表的名称
                localField: "author_id",// article表的 author_id 字段
                foreignField: "_id",// user表的 _id 字段
                as: "user",
            }
        },
    ],
    function (err, docs) {
        console.log(docs[0].cate);
        console.log(docs[0].user);
    }
);
```
